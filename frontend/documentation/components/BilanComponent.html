import { Component, Input } from '@angular/core';
import { CommonModule } from '@angular/common';
import { FormArray, FormBuilder, FormGroup, Validators, ReactiveFormsModule } from '@angular/forms';

@Component({
  selector: 'app-bilan',
  standalone: true,
  imports: [CommonModule, ReactiveFormsModule],
  templateUrl: './bilan.component.html',
  styleUrls: ['./bilan.component.scss']
})
export class BilanComponent {
  @Input() parentForm!: FormGroup;
  isConfirmed = false;

  constructor(private fb: FormBuilder) {}

  get bilanArray(): FormArray {
    return this.parentForm.get('bilan') as FormArray;
  }

  addBilan(): void {
    if (!this.isConfirmed) {
      this.bilanArray.push(this.fb.control('', Validators.required));
    }
  }

  confirmBilans(): void {
    this.isConfirmed = true;
  }

  trackByIndex(index: number): number {
    return index;
  }
}

<div class="bilan-container">
  <div [formGroup]="parentForm">
    <div formArrayName="bilan">
      <div *ngFor="let control of bilanArray.controls; let i = index; trackBy: trackByIndex">
        <input [formControlName]="i" type="text" (keyup.enter)="addBilan()" 
               placeholder="Bilan test {{i + 1}}" />
      </div>
    </div>
    <div class="but" *ngIf="!bilanArray.disabled">
      <button *ngIf="!isConfirmed" type="button" (click)="addBilan()">Ajouter</button>
      <button  *ngIf="!isConfirmed" type="button" (click)="confirmBilans()">Confirmer</button>
    </div>
  </div>
</div>
